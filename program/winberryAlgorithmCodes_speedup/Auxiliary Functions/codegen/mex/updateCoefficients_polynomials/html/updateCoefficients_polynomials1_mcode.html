 6% Updates polynomial coefficients approximating the conditional expectation function in steady state
 7% 
 8% Inputs
 9%   (1) mCoefficients: nEpsilon x nAssets matrix, storing previous iteration's coefficients
10%
11% Outputs
12%   (1) mCoefficientsNew: nEpsilon x nAssets matrix, storing updated coefficients
13%

16% Declare global variables

18%---------------------------------------------------------------
19% Compute current period's savings policy function
20%---------------------------------------------------------------
21
22% Compute conditional expectation
23mConditionalExpectation = exp(mCoefficients * vAssetsPoly');
24
25% Compute target saving
26mAssetsPrimeStar = w * (mmu * (1 - mEpsilonGrid) + (1 - ttau) * mEpsilonGrid) + (1 + r) * mAssetsGrid - ...
27	(mConditionalExpectation .^ (-1 / ssigma));
28
29% Compute actual saving
30mAssetsPrime = max(mAssetsPrimeStar,aaBar);
31mAssetsPrimeGrid = repmat(reshape(mAssetsPrime,1,nState),[nEpsilon 1]);
32
33% Compute next period's polynomials
34mAssetsPrimeZeros = min(max(2 * ((mAssetsPrime - assetsMin) / (assetsMax - assetsMin)) - 1,-1),1); 
35mPolyAssetsPrime = ones(nState,nAssets);
36mPolyAssetsPrime(:,2) = mAssetsPrimeZeros(:);
37for iPower = 3:nAssets
38	mPolyAssetsPrime(:,iPower) = 2 * mAssetsPrimeZeros(:) .* mPolyAssetsPrime(:,iPower-1) - mPolyAssetsPrime(:,iPower-2);
39end
40
41%---------------------------------------------------------------
42% Compute next period's savings policy function
43%---------------------------------------------------------------
44
45% Compute conditional expectation
46mConditionalExpectationPrime = exp(mCoefficients * mPolyAssetsPrime');
47
48% Compute target saving

50	(mConditionalExpectationPrime .^ (-1 / ssigma));
51
52% Compute actual savings
53% mAssetsPrimePrimeGrid = max(mAssetsPrimePrimeStar,aaBar*ones(nEpsilon,nEpsilon*nAssets));
54mAssetsPrimePrimeGrid = max(mAssetsPrimePrimeStar,aaBar);

56%---------------------------------------------------------------
57% Update conditional expectation function
58%---------------------------------------------------------------
59
60% Compute new conditional expectation function

62	mAssetsPrimeGrid - mAssetsPrimePrimeGrid;
63aConditionalExpectationTilde = reshape(bbeta * mEpsilonTransition * ((1 + r) * (mConsumptionPrime .^ (-ssigma))),...

67mConditionalExpectation = zeros(nEpsilon,nAssets);
68for iEpsilon = 1:nEpsilon
69	mConditionalExpectation(iEpsilon,:) = aConditionalExpectationTilde(iEpsilon,iEpsilon,:);
70end
71
72% Update the coefficients
73mCoefficientsNew = zeros(nEpsilon,nAssets);
74for iEpsilon = 1:nEpsilon

76    vCoefficients = vAssetsPoly' * log(mConditionalExpectation(iEpsilon,:))';
77	mCoefficientsNew(iEpsilon,:) = (vCoefficients ./ vAssetsPolySquared)';
78end
